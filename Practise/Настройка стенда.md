Хост - Fedora 43.

Сначала установил виртуалки: Ubuntu, Debian и Windows 11.

На Ubuntu установил Wazuh: Indexer, Server и Dashboard. 

На Debian и Windows 11 агенты с помощью интерфейса Wazuh.

Чтобы виртуалки были доступны с внешней сети, добавить конфиг на хосте:
```
sudo tee /etc/nftables.conf > /dev/null <<EOF
#!/usr/sbin/nft -f

flush ruleset

table inet filter {
    # Трафик к самому хосту Fedora (оставляем базовую защиту)
    chain input {
        type filter hook input priority 0; policy accept;
        ct state established,related accept
        iif "lo" accept
        ip protocol icmp accept
    }

    # Трафик СКВОЗЬ хост (Ноутбук <-> VM и VM <-> Интернет)
    chain forward {
        # Ставим POLICY ACCEPT - это убирает любые ограничения на пересылку
        type filter hook forward priority 0; policy accept;
    }

    chain output {
        type filter hook output priority 0; policy accept;
    }
}

# Таблица NAT (обязательна для интернета в VM)
table ip nat {
    chain postrouting {
        type nat hook postrouting priority 100; policy accept;
        # Разрешаем выход в интернет через основной интерфейс
        ip saddr 192.168.122.0/24 oifname "enp4s0" masquerade
    }
}
EOF

# Выклчить firewalld
sudo systemctl stop firewalld

# Применяем конфигурацию
sudo nft -f /etc/nftables.conf
```

Также на Windows виртуалке нужно сделать:
```
netsh advfirewall firewall add rule name="Allow Ping" dir=in action=allow protocol=icmpv4 powershell Get-NetConnectionProfile | Set-NetConnectionProfile -NetworkCategory Private
```
