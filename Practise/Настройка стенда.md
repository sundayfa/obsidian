Хост - Fedora 43.

Чтобы виртуалки были доступны с внешней сети, прописать в терминале на хосте:
```
# Определяем имя интерфейса автоматически для справки (необязательно)
INTERFACE=$(ip route | grep default | awk '{print $5}')
echo "Ваш сетевой интерфейс: $INTERFACE"

sudo nft flush ruleset

sudo tee /etc/nftables.conf > /dev/null <<EOF
#!/usr/sbin/nft -f

flush ruleset

table inet filter {
    chain input {
        type filter hook input priority 0; policy accept;
    }

    chain forward {
        type filter hook forward priority 0; policy accept;
    }

    chain output {
        type filter hook output priority 0; policy accept;
    }
}

table ip nat {
    chain postrouting {
        type nat hook postrouting priority 100; policy accept;
        # Применяем маскарад для ВСЕГО трафика из сети виртуалок, 
        # уходящего КУДА УГОДНО (кроме самой этой сети)
        ip saddr 192.168.122.0/24 masquerade
    }
}
EOF

sudo nft -f /etc/nftables.conf
```
